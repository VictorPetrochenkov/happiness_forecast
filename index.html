def load_data():
    """    Obr  """
    years = ['2015', '2016', '2017', '2018', '2019']

        #    
        file_name = f"{year}_obr.csv"  #      _1
        file_path = os.path.join(INPUT_DIR, file_name)
        
        if not os.path.exists(file_path):
            #    
            alt_file_name = f"{year}_obr_1.csv"
            alt_file_path = os.path.join(INPUT_DIR, alt_file_name)
            
            if os.path.exists(alt_file_path):

                )
        else:

            print(f" : {file_name}")
        
        dfs.append(df)
    
    return pd.concat(dfs, ignore_index=True)

def train_stacking_model():
    """   -"""
    try:
        print(" ...")

        X = data.drop(['country', 'happiness_score'], axis=1)
        y = data['happiness_score']

            ('rf1', RandomForestRegressor(n_estimators=100, random_state=42)),

        ]
        
        #  -

def load_test_data():
    """    """
    years = ['2015', '2016', '2017', '2018', '2019']
    test_data = {}
    
    for year in years:
        file_path = os.path.join(BASE_DIR, f"Obr/{year}_obr.csv")
        if not os.path.exists(file_path):
            file_path = os.path.join(BASE_DIR, f"Obr/{year}_obr_1.csv")
            
        df = pd.read_csv(file_path)
        test_data[year] = df
    
    return test_data

def evaluate_stacking_model(model):
    """    """
    test_data = load_test_data()
    results = []
    
    for year, df in test_data.items():
        #  
        X = df.drop(['country', 'happiness_score'], axis=1)
        y_true = df['happiness_score']
        countries = df['country']
        
        # 
        y_pred = model.predict(X)
        
        #  
        for i in range(len(df)):
            results.append({
                '': countries.iloc[i],
                '': year,
                '  ': y_true.iloc[i],
                ' ': y_pred[i],
                '': abs(y_true.iloc[i] - y_pred[i])
            })
    
    #  DataFrame  
    results_df = pd.DataFrame(results)
    
    #  
    rmse = np.sqrt(mean_squared_error(
        results_df['  '],
        results_df[' ']
    ))
    r2 = r2_score(
        results_df['  '], 
        results_df[' ']
    )
    
    return results_df, rmse, r2
